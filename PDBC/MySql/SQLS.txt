=========== DATA BASE =================
SHOW DATABASES;         (it shows all the databases)
CREATE DATABASE rays;   (it create new database rays)
DROP DATABASE rays;     (it deletes the database rays)
USE rays;               (from these query we can use these particular database)


=========== DDL & DML ================
CREATE TABLE employee (id INT PRIMARY KEY,name VARCHAR(20),company VARCHAR(30),salary INT);  (it create the table employeee)
INSERT INTO employee VALUES(1,'Ram','Google',89000);       (insert values into the table)
SELECT * FROM employee;                                    (it select all the data from the table)
UPDATE employee SET name = 'Shyam' WHERE id=1;             (it update the existing data)
DELETE FROM employee WHERE id=1;                           (it deletes the data where id=1)
DROP TABLE employee;                                       (it deletes the whole table)


========== SEARCHING =================
SELECT * FROM employee WHERE id = 1;                      (it select data where id = 1)
SELECT * FROM employee WHERE salary > 50000;              (it select data where salary > 5000)
SELECT * FROM employee WHERE name LIKE "s%";              (it select data where in name the first letter is s/S)
SELECT * FROM employee WHERE name LIKE "%r";              (it select data where in name the last letter is r)
SELECT * FROM employee WHERE name LIKE "%p%";             (it select data where in name anywhere the letter is p )
SELECT * FROM employee LIMIT 0,5;                         (it select data from 0 to 4th index)
SELECT * FROM employee ORDER BY salary DESC LIMIT 0,3;    (it arrange the data into descreasing order by salary and shows first 3 data)


========== AGGREGATE FUNCTIONS ============
SELECT COUNT(*) FROM employee;                             (it count the no. of rows )
SELECT SUM(salary) FROM employee;                          (it gives the sum of salary)
SELECT MAX(salary) FROM employee;                          (it gives the max salary)
SELECT MIN(salary) FROM employee;                          (it gives the min salary)
SELECT AVG(salary) FROM employee;                          (it gives the average salary)
SELECT company, COUNT(*) FROM employee GROUP BY company;   (it gives the company and the no. of employee in a particular company)



=========== FOREIGN KEY =======================
CREATE TABLE department(id INT PRIMARY KEY,dept_name VARCHAR(30));                                                                     (it is the parent table)
CREATE TABLE employee (id INT PRIMARY KEY,name VARCHAR(35),dept_id INT, FOREIGN KEY(dept_id) REFERENCES department(id);                (in these table the dept_id is the foreign key that creates by the reference of department table)



=========== JOINS ===========================
SELECT * FROM employee INNER JOIN department ON employee.dept_id = department.id;           (it shows only those data which is common in both tables)
SELECT * FROM employee LEFT JOIN department ON employee.dept_id = department.id;            (it shows all the data of left table and the common data in the right table)
SELECT * FROM employee RIGHT JOIN department ON employee.dept_id = department.id;           (it shows all the data of right table and the common data in the left table)
SELECT * FROM employee LEFT JOIN department ON employee.dept_id = department.id UNION SELECT * FROM employee RIGHT JOIN department ON employee.dept_id = department.id;         (it shows all data of both the tables)
